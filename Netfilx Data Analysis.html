def catigorize_col(df, col, labels):
    """
    Categorizes a column in a DataFrame based on its quartiles.

    Args:
        df (pd.DataFrame): The DataFrame containing the column to be categorized.
        col (str): The name of the column to categorize.
        labels (list): A list of labels to assign to the quartiles.

    Returns:
        pd.DataFrame: The DataFrame with the categorized column.
    """
    # Setting the edges to cut the column accordingly
    edges = [
        df[col].describe()['min'],
        df[col].describe()['25%'],
        df[col].describe()['50%'],
        df[col].describe()['75%'],
        df[col].describe()['max']
    ]
    
    # Categorizing the column
    df[col] = pd.cut(df[col], edges, labels=labels, duplicates='drop')
    return df
